empire_of_china_nomadic_border = {
	potential = {
		has_dlc = "Mandate of Heaven"
		is_emperor_of_china = yes
	}
	can_start = {
		has_any_disaster = no
		any_neighbor_country = {
			has_reform = steppe_horde
			total_own_and_non_tributary_subject_development = 300
			is_subject = no
			NOT = {
				truce_with = ROOT
			}
			NOT = {
				alliance_with = ROOT
			}
		}
	}
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				any_neighbor_country = {
					has_reform = steppe_horde
					total_own_and_non_tributary_subject_development = 300
					NOT = {
						truce_with = ROOT
					}
					NOT = {
						alliance_with = ROOT
					}
					is_subject = no
				}
			}
		}
	}
	progress = {
		modifier = {
			factor = 1
			any_neighbor_country = {
				has_reform = steppe_horde
				total_own_and_non_tributary_subject_development = 300
				is_subject = no
				NOT = {
					truce_with = ROOT
				}
				NOT = {
					alliance_with = ROOT
				}
			}
		}
		modifier = {
			factor = 1.5
			any_neighbor_country = {
				has_reform = steppe_horde
				total_own_and_non_tributary_subject_development = 300
				war_with = ROOT
				war_score_against = {
					who = ROOT
					value = 10
				}
			}
		}
		modifier = {
			factor = 3
			any_neighbor_country = {
				has_reform = steppe_horde
				total_own_and_non_tributary_subject_development = 300
				war_with = ROOT
				war_score_against = {
					who = ROOT
					value = 25
				}
			}
		}
		modifier = {
			factor = -3
			any_neighbor_country = {
				has_reform = steppe_horde
				war_with = ROOT
				total_own_and_non_tributary_subject_development = 300
				ROOT = {
					war_score_against = {
						who = PREV
						value = 25
					}
				}
			}
		}
	}
	can_end = {
		OR = {
			is_emperor_of_china = no
			NOT = {
				any_neighbor_country = {
					has_reform = steppe_horde
					total_own_and_non_tributary_subject_development = 300
					NOT = {
						alliance_with = ROOT
					}
					is_subject = no
				}
			}
		}
	}
	modifier = {
		imperial_mandate = -0.3
		reduced_liberty_desire = -25
		land_morale = -0.15
	}
	on_start = empire_of_china_nomadic_border_events.1
	on_end = empire_of_china_nomadic_border_events.2
	on_monthly = {
		events = {
			mch_missions.5
			ming_crisis.3
			ming_crisis.4
			ming_crisis.5
		}
		random_events = {
		}
	}
	on_start_effect = {
		hidden_effect = {
			clr_country_flag = empire_of_china_nomadic_border_started_ticker
			clr_country_flag = empire_of_china_nomadic_border_further_ticker
			every_country = {
				limit = {
					has_country_flag = empire_of_china_nomadic_border_cause
				}
				clr_country_flag = empire_of_china_nomadic_border_cause
			}
		}
	}
	on_progress_effect = {
		if = {
			limit = {
				NOT = {
					has_country_flag = empire_of_china_nomadic_border_started_ticker
				}
				check_variable = {
					which = empire_of_china_nomadic_border_progress
					value = 10
				}
			}
			set_country_flag = empire_of_china_nomadic_border_started_ticker
			every_neighbor_country = {
				limit = {
					has_reform = steppe_horde
					total_own_and_non_tributary_subject_development = 300
					NOT = {
						alliance_with = ROOT
					}
					is_subject = no
				}
				set_country_flag = empire_of_china_nomadic_border_cause
				country_event = {
					id = empire_of_china_nomadic_border_events.3
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = empire_of_china_nomadic_border_further_ticker
				}
				check_variable = {
					which = empire_of_china_nomadic_border_progress
					value = 75
				}
			}
			set_country_flag = empire_of_china_nomadic_border_further_ticker
			every_neighbor_country = {
				limit = {
					has_reform = steppe_horde
					total_own_and_non_tributary_subject_development = 300
					NOT = {
						alliance_with = ROOT
					}
					is_subject = no
				}
				country_event = {
					id = empire_of_china_nomadic_border_events.4
				}
			}
		}
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = empire_of_china_nomadic_border_progress
						value = 5
					}
				}
				has_country_flag = empire_of_china_nomadic_border_started_ticker
			}
			every_country = {
				limit = {
					has_country_flag = empire_of_china_nomadic_border_cause
				}
				country_event = {
					id = empire_of_china_nomadic_border_events.5
				}
			}
		}
	}
}
